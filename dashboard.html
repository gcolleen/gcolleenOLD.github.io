```{r}
data("nyc_airbnb")

airbnb <- nyc_airbnb  %>%
  rename(borough = neighbourhood_group) %>%
  mutate(rating = review_scores_location/2) %>%
  select(borough, neighbourhood, rating, price, room_type, lat, long) %>%
  filter(
    borough == "Manhattan", price %in% 100:500,
    room_type == "Entire home/apt"
  ) %>% drop_na(rating)
```
